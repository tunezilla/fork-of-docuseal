```
DEFAULT_USER_FIRST_NAME=First
DEFAULT_USER_LAST_NAME=LAST
DEFAULT_USER_EMAIL=sample@localhost
DEFAULT_USER_PASSWORD=default
DEFAULT_USER_APP_URL=http://localhost:3000
```
